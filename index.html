Start Database Command (Termux only)

mysqld_safe

Enter to Database

mariadb -u root

Use this command for password

mariadb -u root -p

Show existing database

SHOW DATABASES;

Create a database
CREATE DATABASE db_name;
Use a database
USE db_name
Create a table

CREATE TABLE db_name (Id datatype(length), Name datatype(length));

CREATE TABLE db_name (Id INT(16), Name VARCHAR(50));
Insert a value to a Table

CREATE TABLE table_name VALUES (firstColumn, secondColumn);

INSERT INTO db_name VALUES ("This is", "an example");
Alter Table Add Column

ALTER TABLE table_name ADD COLUMN column_name datatype(length);

ALTER TABLE db_name ADD COLUMN DateRegistered VARCHAR(50);
Alter Table Delete Column

ALTER TABLE table_name DROP COLUMN column_name

ALTER TABLE db_name DROP COLUMN Address
Alter Table Add in First Column
ALTER TABLE db_name ADD COLUMN Id INT(16) FIRST
Alter Table Add in Specific Column
ALTER TABLE db_name ADD COLUMN Name VARCHAR(50) BEFORE Id;
Alter Table Add Primary Key

Primary Key is used for preventing existing data from being submitted twice

ALTER TABLE db_name ADD CONSTRAINT PRIMARY KEY(column_name);
